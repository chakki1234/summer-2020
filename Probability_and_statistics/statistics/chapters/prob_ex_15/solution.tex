\begin{split}
f_0 &= frequency\ of\ class\ \\ &before\ modal\ class
\end{split}\\
\begin{split}
f_2 &= frequency\ of\ class\ \\ &after\ modal\ class
\end{split}
\end{align*}

\begin{align}
Modal\ class &= 40 - 50\\ 
l &= 40\\
h &= 10 - 0 = 10\\
f_1 &= 20\\
f_0 &= 12\\
f_2 &= 11\\
Mode &= 40 + \frac{20-12}{2\brak{20} - 12- 11} \times 10\\
&= 44.7
\end{align}
